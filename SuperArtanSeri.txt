#include <stdio.h>
#include <stdlib.h>
#include <math.h>


int main(int argc, char *argv[]) {
	printf("Sayiyi giriniz: ");
	int sayi,i=0,dizi[10],a;
	scanf("%d",&sayi);
	while(sayi!=0 && sayi!=1){
		dizi[i]=sayi%2;
		sayi=sayi/2;
		i++;
	}
	if(sayi!=0){
		dizi[i]=1;
	}
	if(sayi==0)printf("0\n");
	
	for(a=i;a>=0;a--){
		dizi[i]==1 ? printf("2^%d ",i):printf("");
		i!=0 && dizi[i-1]==1 ? printf("+ "): printf("");
		i--;
	}
	
	
	return 0;
}
