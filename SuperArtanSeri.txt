#include <stdio.h>
#include <stdlib.h>

int main(int argc, char *argv[]) {
	printf("Sayiyi giriniz: ");
	int sayi,i=0,dizi[30],a,x;;
	scanf("%d",&sayi);
	printf("Taban degerini giriniz: ");
	scanf("%d",&x);
	while(sayi!=0 && sayi!=1){
		dizi[i]=sayi%x;
		sayi=sayi/x;
		i++;
	}
	if(sayi!=0){
		dizi[i]=1;
	}
	if(sayi==0)printf("0\n");
	
	for(a=i;a>=0;a--){
		dizi[i]==1 ? printf("%d^%d ",x,i):printf("");
		i!=0 && dizi[i-1]==1 ? printf("+ "): printf("");
		i--;
	}
	
	
	return 0;
}
